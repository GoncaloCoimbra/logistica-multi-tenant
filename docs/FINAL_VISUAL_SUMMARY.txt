 CREATED DATA
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  History (AuditLog):
  ├─ Table: AuditLog
  ├─ Records: action, entity, entityId, userId, companyId, ip, timestamp
  └─ Access: GET /api/audit-logs

  Notifications:
  ├─ Table: Notification
  ├─ Stores: title, content, isRead, userId, companyId, timestamp
  └─ Access: GET /api/notifications


🎯 MAIN ENDPOINTS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  HISTORY:
  GET  /api/audit-logs              → List
  GET  /api/audit-logs/entity/:...  → By entity
  GET  /api/audit-logs/user/:...    → By user

  NOTIFICATIONS:
  GET  /api/notifications           → List
  GET  /api/notifications/unread    → Unread
  PATCH /api/notifications/:id/read → Mark read
  DELETE /api/notifications/:id     → Delete

  PRODUCTS (Triggers notifications):
  POST /api/products                → Create + notification
  PUT  /api/products/:id/status     → Update + notification
