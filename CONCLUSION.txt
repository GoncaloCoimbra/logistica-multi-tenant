 MAIN ENDPOINTS

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

HISTORY:
  GET  /api/audit-logs              → List all
  GET  /api/audit-logs/entity/:entity/:id → By entity
  GET  /api/audit-logs/user/:userId → By user
  GET  /api/audit-logs/stats        → Statistics

NOTIFICATIONS:
  GET  /api/notifications           → List all
  GET  /api/notifications/unread    → Unread
  GET  /api/notifications/unread/count → Count unread
  PATCH /api/notifications/:id/read → Mark as read
  PATCH /api/notifications/read-all → Mark all as read
  DELETE /api/notifications/:id     → Delete

PRODUCTS (Triggers notifications):
  POST /api/products                → Creates + notification
  PUT  /api/products/:id/status     → Updates status + notification
